// Firestore Security Rules for Leduc Islamic Association
// Copy these rules to: Firebase Console > Firestore > Rules
//
// These rules allow:
// - Anyone to READ settings (so the website can load prayer times, etc.)
// - Only authenticated users to WRITE settings (admin only)

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /settings/{document} {
      allow read: if true;
      allow write: if request.auth != null;
    }
  }
}
